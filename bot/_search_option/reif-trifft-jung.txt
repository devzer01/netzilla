<script type="text/javascript">
function disableOpt(sex) {
	if(sex == '2') {
		$("optgroup[label='Female'] > option")
			.removeAttr("disabled")
			.eq(0).prop('selected',true);
		$("optgroup[id='ageToMale'] > option")
			.eq(0).prop('selected',true);
		$("optgroup[label='Male'] > option").attr("disabled","disabled");
	} else {
		$("optgroup[label='Male'] > option")
			.removeAttr("disabled")
			.eq(0).prop('selected',true);
		$("optgroup[id='ageToFemale'] > option")
			.eq(0).prop('selected',true);
		$("optgroup[label='Female'] > option").attr("disabled","disabled");
	}
}

$(function(){
	var sex = $("#gender option:selected").val();
	disableOpt(sex);
	
	$("#gender").change(function(){
		var sex = $(this).val();
		disableOpt(sex);
	});
	
	
});
</script>
