<script>

function changeMeetmeCountry()
{
	if($('#country').val()=="United Kingdom")
	{
		$('#start_city1').attr('name','start_city');
		$('#start_city2').attr('name','start_city2');
		$('#start_city1').show();
		$('#start_city2').hide();
	}
	else if($('#country').val()=="Germany")
	{
		$('#start_city1').attr('name','start_city1');
		$('#start_city2').attr('name','start_city');
		$('#start_city1').hide();
		$('#start_city2').show();
	}
}
</script>
