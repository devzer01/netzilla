<script>
function changeFlirtFeverPLZ()
{
	if($('#country').val()=="1")
	{
		if($('#distance').val()=="")
		{
			$('#plz1').attr('name','plz1');
			$('#plz2').attr('name','plz2');
			$('#plz3').attr('name','plz3');
			$('#plz4').attr('name','plz4');
			$('#plz1').hide();
			$('#plz2').hide();
			$('#plz3').hide();
			$('#plz4').hide();
		}
		else if(parseInt($('#distance').val())<500)
		{
			$('#plz1').attr('name','start_plz');
			$('#plz2').attr('name','plz2');
			$('#plz3').attr('name','plz3');
			$('#plz4').attr('name','plz4');
			$('#plz1').show();
			$('#plz2').hide();
			$('#plz3').hide();
			$('#plz4').hide();
		}
		else
		{
			$('#plz1').attr('name','plz1');
			$('#plz2').attr('name','start_plz');
			$('#plz3').attr('name','plz3');
			$('#plz4').attr('name','plz4');
			$('#plz1').hide();
			$('#plz2').show();
			$('#plz3').hide();
			$('#plz4').hide();
		}
	}
	else if($('#country').val()=="2")
	{
		$('#plz1').attr('name','plz1');
		$('#plz2').attr('name','plz2');
		$('#plz3').attr('name','start_plz');
		$('#plz4').attr('name','plz4');
		$('#plz1').hide();
		$('#plz2').hide();
		$('#plz3').show();
		$('#plz4').hide();
	}
	else
	{
		$('#plz1').attr('name','plz1');
		$('#plz2').attr('name','plz2');
		$('#plz3').attr('name','plz4');
		$('#plz4').attr('name','start_plz');
		$('#plz1').hide();
		$('#plz2').hide();
		$('#plz3').hide();
		$('#plz4').show();
	}
}
</script>
