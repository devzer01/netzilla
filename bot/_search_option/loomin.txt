<script>
function changeKTOSexyPLZ()
{
	if(parseInt($('#distance').val())<=100)
	{
		$('#plz1').attr('name','start_plz');
		$('#plz2').attr('name','plz2');
		$('#plz1').show();
		$('#plz2').hide();
	}
	else
	{
		$('#plz1').attr('name','plz1');
		$('#plz2').attr('name','start_plz');
		$('#plz1').hide();
		$('#plz2').show();
	}
}

function checkOnline()
{
	if($('#online').attr('checked'))
		$('#region').hide();
	else
		$('#region').show();
}
</script>
