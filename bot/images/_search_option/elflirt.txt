<script>
$(document).ready(function() {

	 $('#search').change(function(){
		var search = $('#search :selected').val();

		if(search == 1){
		   
		    $("#gender_area").show();
		    $("#age_area").hide();
		    $("#land_area").hide();
		    $("#foto_area").hide();
		    $("#searchof_area").hide();
		    $("#stand_area").hide();

		    $("#age_area").empty();		   

		}else if(search == 2){

		    $("#gender_area").show();
		    $("#age_area").hide();
		    $("#land_area").hide();
		    $("#foto_area").hide();
		    $("#searchof_area").hide();
		    $("#stand_area").hide();

		}else{

		    $("#gender_area").show();
		    $("#age_area").show();
		    $("#land_area").show();
		    $("#foto_area").show();
		    $("#searchof_area").show();
		    $("#stand_area").show();

		}
	 });

});
</script>
